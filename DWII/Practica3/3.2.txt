JS
const lienzo = document.querySelector('#lienzoTexto');
const texto = document.querySelector('#textoPrincipal');

const inputTexto = document.querySelector('#inputTexto');
const sliderTamaño = document.querySelector('#sliderTamaño');
const botonesColor = document.querySelectorAll('.boton-color'); 


inputTexto.addEventListener('input', function() {
  texto.textContent = inputTexto.value.toUpperCase(); 
});


sliderTamaño.addEventListener('input', function() {
  texto.style.fontSize = sliderTamaño.value + 'px';
});


botonesColor.forEach(function(boton) {
  
  boton.addEventListener('click', function() {
    const codigoColor = boton.getAttribute('data-color');
    
    lienzo.style.setProperty('background-color', codigoColor, 'important');    
    if (codigoColor === '#212529') {
      texto.style.color = '#ffffff';
    } else {
      texto.style.color = '#212529';
    }
  });
  
});
