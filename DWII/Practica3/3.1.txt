JS

// PASO 1: 
const boton = document.querySelector('#botonInterruptor');
const lienzo = document.querySelector('.contenedor-interactivo');

// PASO 2:
boton.addEventListener('click', function() {
    
    // PASO 3: 
    lienzo.classList.toggle('modo-oscuro-activo');
    
    // PASO 4: 
    if (lienzo.classList.contains('modo-oscuro-activo')) {
        boton.textContent = "Apagar Modo Oscuro";
        boton.className = "btn btn-light btn-lg rounded-pill px-4"; // Cambia el color del botón
    } else {
        boton.textContent = "Prender Modo Oscuro";
        boton.className = "btn btn-primary btn-lg rounded-pill px-4";
    }
    
});
